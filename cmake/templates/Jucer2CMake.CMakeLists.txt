cmake_minimum_required(VERSION 3.0)


list(APPEND CMAKE_MODULE_PATH "@cmake_dir@")
include(Reprojucer)


get_filename_component(@jucer_file_name_var@_FILE
  "@jucer_file@"
  ABSOLUTE CACHE
)

if(NOT EXISTS "${@jucer_file_name_var@_FILE}")
  message(FATAL_ERROR "No such file: ${@jucer_file_name_var@_FILE}")
endif()

get_filename_component(@jucer_file_name_var@_DIR
  "${@jucer_file_name_var@_FILE}"
  DIRECTORY
)


jucer_project_begin(
  PROJECT_NAME "HelloWorld"
  PROJECT_VERSION "1.0.0"
  COMPANY_NAME "ROLI Ltd."
  # COMPANY_WEBSITE
  # COMPANY_EMAIL
  PROJECT_TYPE "GUI Application"
  BUNDLE_IDENTIFIER "com.roli.jucehelloworld"
  BINARYDATACPP_SIZE_LIMIT "Default"
  # BINARYDATA_NAMESPACE
  # PREPROCESSOR_DEFINITIONS
  PROJECT_ID "tTAKTK1s"
)

jucer_project_files("HelloWorld/Source"
  "${@jucer_file_name_var@_DIR}/Source/Main.cpp"
  "${@jucer_file_name_var@_DIR}/Source/MainComponent.cpp"
  "${@jucer_file_name_var@_DIR}/Source/MainComponent.h"
)

  juce_core
  PATH "${@jucer_file_name_var@_DIR}/../../modules"
  # JUCE_FORCE_DEBUG

  # JUCE_DONT_AUTOLINK_TO_WIN32_LIBRARIES
  # JUCE_INCLUDE_ZLIB_CODE
  # JUCE_USE_CURL

  # JUCE_ALLOW_STATIC_NULL_VARIABLES
)

  PATH "${@jucer_file_name_var@_DIR}/../../modules"

  # JUCE_USE_COREIMAGE_LOADER

)
